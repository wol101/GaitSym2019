132 void GLAPIENTRY
133 gluPerspective(double fovy, double aspect, double zNear, double zFar)
134 {
135  double m[4][4];
136  double sine, cotangent, deltaZ;
137  double radians = fovy / 2 * __glPi / 180;
138 
139  deltaZ = zFar - zNear;
140  sine = sin(radians);
141  if ((deltaZ == 0) || (sine == 0) || (aspect == 0))
142  {
143  return;
144  }
145  cotangent = COS(radians) / sine;
146 
147  __gluMakeIdentityd(&m[0][0]);
148  m[0][0] = cotangent / aspect;
149  m[1][1] = cotangent;
150  m[2][2] = -(zFar + zNear) / deltaZ;
151  m[2][3] = -1;
152  m[3][2] = -2 * zNear * zFar / deltaZ;
153  m[3][3] = 0;
154  glMultMatrixd(&m[0][0]);
155 }

177 gluLookAt(double eyex, double eyey, double eyez, double centerx,
178  double centery, double centerz, double upx, double upy,
179  double upz)
180 {
181  float forward[3], side[3], up[3];
182  float m[4][4];
183 
184  forward[0] = centerx - eyex;
185  forward[1] = centery - eyey;
186  forward[2] = centerz - eyez;
187 
188  up[0] = upx;
189  up[1] = upy;
190  up[2] = upz;
191 
192  normalize(forward);
193 
194  /* Side = forward x up */
195  cross(forward, up, side);
196  normalize(side);
197 
198  /* Recompute up as: up = side x forward */
199  cross(side, forward, up);
200 
201  __gluMakeIdentityf(&m[0][0]);
202  m[0][0] = side[0];
203  m[1][0] = side[1];
204  m[2][0] = side[2];
205 
206  m[0][1] = up[0];
207  m[1][1] = up[1];
208  m[2][1] = up[2];
209 
210  m[0][2] = -forward[0];
211  m[1][2] = -forward[1];
212  m[2][2] = -forward[2];
213 
214  glMultMatrixf(&m[0][0]);
215  glTranslated(-eyex, -eyey, -eyez);
216 }
