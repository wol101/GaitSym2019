366 // returns the difference between the target actual value and the desired value (actual - desired)
367 double DataTargetScalar::calculateError(double time)
368 {
369  size_t index, indexNext;
370  auto lowerBound = std::lower_bound(targetTimeList()->begin(), targetTimeList()->end(), time);
371  auto upperBound = std::upper_bound(targetTimeList()->begin(), targetTimeList()->end(), time);
372  if (lowerBound == targetTimeList()->begin()) // time <= lowest value in the list
373  {
374  index = 0;
375  indexNext = index;
376  }
377  else if (upperBound == targetTimeList()->end()) // time > highest value in the list
378  {
379  index = targetTimeList()->size() - 1;
380  indexNext = index;
381  }
382  else
383  {
384  index = std::distance(targetTimeList()->begin(), lowerBound) - 1; // subtracting 1 because lower bound gives an index 1 higher than expected (IMO)
385  indexNext = std::min(index + 1, targetTimeList()->size() - 1);
386  }
387 
388  double value = GSUtil::Interpolate((*targetTimeList())[size_t(index)], m_ValueList[size_t(index)], (*targetTimeList())[indexNext], m_ValueList[indexNext], time);
389  return calculateErrorScore(value);
390 }
